工厂类

工厂类、单例模式

  • 通过专门定义一个类来负责创建其他类的实例，被创建的实例通常都具有共同的父类 并且重写父类方法

    class  Operation:
    def __init__(self,numA,numB):
        self.numA=numA
        self.numB=numB
    def getResult(self):
        print('计算2个操作数的运算结果')

class Add(Operation):
    def __init__(self,numA,numB):
        super(Add, self).__init__(numA,numB)
    def getResult(self):
        return self.numA+self.numB
class Minus(Operation):
    def __init__(self,numA,numB):
        super(Minus, self).__init__(numA,numB)
    def getResult(self):
        return self.numA-self.numB

class Factory:            #根据不同的参数，创建不同的对象的函数
    def create_class(self,ope,numA,numB):
        if ope == '+':
            q = Add(numA, numB)
        elif ope == '-':
            q = Minus(numA, numB)
        else:
            print('输入错误！')
        return q

if __name__=='__main__':
    factory=Factory()             #创建一个工厂对象
    numA=int(input('请输入第一个数:'))
    numB=int(input('请输入第二个数:'))
    ope=input('请输入运算符:')
    q=factory.create_class(ope,numA,numB)  #代表一个计算器
    print('计算结果:',q.getResult())

class dvd:
    def __init__(self,name,price,status):
        self.name=name
        self.price=price
        self.status=status
    def __str__(self):
        stat='已借出'
        if self.status==1:
            stat='未借出'
        return '名称：《%s》  单价：%d  状态：%s'%(self.name,self.price,stat)
class DvdManage:
    dvds=[]  #存储  多个dvds，每个元素都是一个DVD
    def init(self):
        self.dvds.append(dvd('光辉岁月',10,0))
        self.dvds.append(dvd('大海日出',12,1))
        self.dvds.append(dvd('你好小明',20,1))
    def Menu(self):
        self.init()
        while(True):
            print('1.查询所有DVD')
            print('2.增加DVD')
            print('3.借出DVD')
            print('4.归还DVD')
            print('5.退出')
            choose=int(input('请选择操作项:'))
            if choose==1:
                self.showAlldvd()
            elif choose==2:
                self.insertDvd()
            elif choose==3:
                self.lend()
            elif choose==4:
                self.backDvd()
            elif choose==5:
                print('5.谢谢使用！')
                break
    def showAlldvd(self):
        for d in self.dvds:
            print(str(d))
    def insertDvd(self):
        name = input('请输入DVD的名称:')
        ret=self.checkDvd(name)
        if ret!=None:
            print('DVD已存在!')
        else:
            price=int(input('请输入价格:'))
            new_dvd=dvd(name,price,1)
            self.dvds.append(new_dvd)
            print('存放成功!')
    def lend(self):
        name = input('请输入DVD的名称:')
        ret = self.checkDvd(name)
        if ret != None:
            if ret.status==0:
                print('DVD已借出!')
            else:
                days=int(input('请输入借出天数:'))
                ret.status=0
                print('借出%d天，应收%d'%(days,ret.price*days))
        else:
            print('DVD不存在!')
    def backDvd(self):
        name=input('请输入DVD的名称:')
        ret=self.checkDvd(name)
        if ret==None:
            print('DVD不存在!')
        else:
            if ret.status==1:
                print('DVD未借出!')
            else:
                days=int(input('请输入借出天数:'))
                y_money=ret.price * days
                print('借出%d天，应收%d' % (days, ret.price * days))
                money=0
                while True:
                    money=int(input('实收:'))
                    if money<y_money:
                        print('不够啊!')
                    else:
                        break
                if money>=y_money:
                    print('找零:%d'%(money-y_money))
                ret.status=1
                print('归还成功!')
    def checkDvd(self,name):
        for d in self.dvds:
            if d.name==name:
                return d       #返回当前对象---内存地址
        else:
            return None

封装、继承、多态

封装

  • 隐藏内部实现，提供外部接口
    两个步骤：属性私有化——set\get函数写方法

    class Student:
    def __init__(self,name,age):
        self.name=name
        self.__age=age
    @property
    def age(self):
        return self.__age
    @age.setter
    def age(self,value):
        if value<0 or value>100:
            self.__age=18
        else:
            self.__age=value

继承

  • 继承：子类继承父类，子类可以使用父类的属性和方法，简化代码
    子类继承父类，父类派生子类（父类又叫基类，子类又叫派生类）
    继承指出两个类符合“A是B”的关系

    class Vehicle:         #交通工具类
    def __init__(self,name):
        self.name=name

class Truck(Vehicle):    #卡车，属性：载重
    #当子类继承父类时，子类的构造方法应该包含父类和子类共有的属性，如name
    #在子类的初始化方法中，将父类的属性传递给父类，子类的属性赋值给子类
    #在子类的初始化方法中，调用父类的属性有四种方法：
    # 1，父类名.属性 2，self.属性 3，父类名.__init__(self,属性) 4，super(子类名,self).__init__(属性)
    def __init__(self,name,weight):
        #Vehicle.name=name                 #1，父类名.属性
        #self.name=name                    #2，self.属性
        #Vehicle.__init__(self,name)       #3，父类名.__init__(self,属性)
        super(Truck,self).__init__(name)   #4，super(子类名,self).__init__(属性)
        self.weight=weight
        print('卡车被初始化！')
    def run(self):        #卡车，重写行驶方法
        #super().show() #调用父类方法
        #Vehicle.show(self)
        super(Truck,self).show()
        print('%s卡车载重%d吨在行驶...'%(self.name,self.weight))

  • 三代继承：子类初始化方法需祖父、父类及自己的属性，可以调用父类的初始化方法传参，可以重写父类的方法，构造的顺序依然先构造祖父类，再构造父类，最后构造自己

  • 多继承：一个类同时继承多个父类，class(A,B),当AB均有相同方法，而子类又重写时，调用子类的方法；如果子类没有方法，则调用优先继承的类

    多态
  • 封装可以隐藏实现细节，使得代码模块化；继承可以扩展已存在的代码模块（类）；它们的目的都是为了——代码重用。而多态则是为了实现另一个目的——接口重用！多态的作用，就是为了类在继承和派生的时候，保证使用“家谱”中任一类的实例的某一属性时的正确调用

  • 多态就是多种表现形态的意思。它是一种机制、一种能力，而非某个关键字。它在类的继承中得以实现，在类的方法调用中得以体现。多态意味着变量并不知道引用的对象是什么，根据引用对象的不同表现不同的行为方式

print(list2)      #result:['Happy', 'New', 'Year', '2018']
list2.insert(3,'2018')        #插入到列表相应下标位置,超出范围则在尾部添加
print(list2)      #result:['Happy', 'New', 'Year', '2018']

list3=['百度','小米','阿里']
list4=['李桑','雷军','马云'] #两个列表可以进行相加 
print(list3+list4)            #result:['百度', '小米', '阿里', '李桑', '雷军', '马云']
print(list3)                  #result:['百度', '小米', '阿里']

list3.extend(list4)           #列表1.extend(列表2)，是把列表2迭代后追加到列表1，extend()没有返回值
print(list3)                  #result:['百度', '小米', '阿里', '李桑', '雷军', '马云']

列表的函数

  • len() 列表元素个数

    list1=['a','b','c','d','e']
print(len(list1))           #result:5

  • max()、min() 返回列表元素最大值、最小值;字母以ASCII为依据

    print(max(list1))           #result:e
print(min(list1))           #result:a

  • list()将元组转换成列表

    a=('a','b','c')             #定义一个元组
print(list(a))              #result:['a', 'b', 'c']

  • append()增加到列表尾部

  • insert(index,obj)插入到列表相应下标位置,超出范围则在尾部添加
  • extend()列表1.extend(列表2)，是把列表2迭代后追加到列表1
  • pop()默认删除最后一个,也可以指定删除的下标，返回删除值
  • remove()删除括号中的内容，但只删除一次

  • reverse()反向列表中的元素

    list2=['a','b','c','d','e']
list2.reverse()
print(list2)                  #result:['e', 'd', 'c', 'b', 'a']

  • sort([func])排序，默认升序

    list3=[5,6,4,2,9]
list3.sort()
print(list3)                  #result:[2, 4, 5, 6, 9]

  • clear() 清空列表

    list4=['a','b','c','d','e']
list4.clear()
print(list4)                  #result:[]

  • copy()复制列表，在内存中又开辟了一个空间；通过等号复制副本，在内存中指向原来的地址

    list5=['a','b','c']
list6=list5.copy()
print(list6)                  #result:['a','b','c']

  • eval()将元组、字典、列表类型的字符串转为元组、字典、列表,可以把list,tuple,dict和string相互转化

    a="[[1,2], [3,4], [5,6]]"        #字符串转换成列表
m=eval(a)
print(type(m))                  #result:<class 'list'>
print(m)                        #result:[[1, 2], [3, 4], [5, 6]]

a="{'name':'Liu','age':'25'}"   #字符串转换成字典
m=eval(a)
print(type(m))                  #result:<class 'dict'>
print(m)                        #result:{'name': 'Liu', 'age': '25'}

a="('a','b','c')"               #字符串转换成元组
m=eval(a)
print(type(m))                  #result:<class 'tuple'>
print(m)                        #result:('a', 'b', 'c')
