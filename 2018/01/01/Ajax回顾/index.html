Ajax回顾

Ajax回顾

简介


Asynchronous Javascript And XML，异步的JavaScript和XML。通过在后台与服务器进行少量数据交换，AJAX可以使网页实现异步更新。即在不重新加载整个网页的情况下，对网页的某部分进行更新。

以封装一个请求为例

function ajax(url,funSucc,funFail){
    //1.创建ajax对象
    var xhr=new XMLHttpRequest()
    ////兼容IE6，IE5
    if(window.XMLHttpRequest){
        var xhr=new XMLHttpRequest()
    }else{
        var xhr=new ActiveXObject("Microsoft.XMLHttp")    
    }
    //2.连接服务器，open(请求类型,文件路径,是否异步)
    //如果是post请求，则需要手动设置请求消息头，如下
    xhr.open('post',url,true);
    xhr.setRequestHeader("Content-type","application/x-www-form-urlencoded");
    //如果是get请求，如下
    xhr.open('get',url,true)
    //3.发送请求，send方法内传递的是提交参数，没有可为null
    xhr.send(null)
    //4.接收数据，为Ajax对象的onreadystatechange事件设定响应函数
    xhr.onreadystatechange=function(){
        //readyState 请求状态:
            0尚未初始化
            1正在发送请求
            2请求完成
            3请求成功，正在接受数据
            4接收数据成功
        if(xhr.readyState==4){
            //status 服务器返回的http请求响应值常用的有：
                200表示请求成功
                202请求被接受但处理未完成
                400错误的请求
                404资源未找到
                500内部服务器错误

            if(xhr.status==200){
                //responseText 服务器返回的文本
                funSucc(xhr.responseText)
            }else{
                //statusText 服务器返回的请求响应文本
                funFail(xhr.statusText)
            }
        }
    }
}
