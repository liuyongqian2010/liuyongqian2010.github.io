DVD管理系统

DVD管理系统
class dvd:
    def __init__(self,name,price,status):
        self.name=name
        self.price=price
        self.status=status
    def __str__(self):
        stat='已借出'
        if self.status==1:
            stat='未借出'
        return '名称：《%s》  单价：%d  状态：%s'%(self.name,self.price,stat)
class DvdManage:
    dvds=[]  #存储  多个dvds，每个元素都是一个DVD
    def init(self):
        self.dvds.append(dvd('光辉岁月',10,0))
        self.dvds.append(dvd('大海日出',12,1))
        self.dvds.append(dvd('你好小明',20,1))
    def Menu(self):
        self.init()
        while(True):
            print('1.查询所有DVD')
            print('2.增加DVD')
            print('3.借出DVD')
            print('4.归还DVD')
            print('5.退出')
            choose=int(input('请选择操作项:'))
            if choose==1:
                self.showAlldvd()
            elif choose==2:
                self.insertDvd()
            elif choose==3:
                self.lend()
            elif choose==4:
                self.backDvd()
            elif choose==5:
                print('5.谢谢使用！')
                break
    def showAlldvd(self):
        for d in self.dvds:
            print(str(d))
    def insertDvd(self):
        name = input('请输入DVD的名称:')
        ret=self.checkDvd(name)
        if ret!=None:
            print('DVD已存在!')
        else:
            price=int(input('请输入价格:'))
            new_dvd=dvd(name,price,1)
            self.dvds.append(new_dvd)
            print('存放成功!')
    def lend(self):
        name = input('请输入DVD的名称:')
        ret = self.checkDvd(name)
        if ret != None:
            if ret.status==0:
                print('DVD已借出!')
            else:
                days=int(input('请输入借出天数:'))
                ret.status=0
                print('借出%d天，应收%d'%(days,ret.price*days))
        else:
            print('DVD不存在!')
    def backDvd(self):
        name=input('请输入DVD的名称:')
        ret=self.checkDvd(name)
        if ret==None:
            print('DVD不存在!')
        else:
            if ret.status==1:
                print('DVD未借出!')
            else:
                days=int(input('请输入借出天数:'))
                y_money=ret.price * days
                print('借出%d天，应收%d' % (days, ret.price * days))
                money=0
                while True:
                    money=int(input('实收:'))
                    if money<y_money:
                        print('不够啊!')
                    else:
                        break
                if money>=y_money:
                    print('找零:%d'%(money-y_money))
                ret.status=1
                print('归还成功!')
    def checkDvd(self,name):
        for d in self.dvds:
            if d.name==name:
                return d       #返回当前对象---内存地址
        else:
            return None
