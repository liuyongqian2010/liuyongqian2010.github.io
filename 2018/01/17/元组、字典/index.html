元组、字典

元组、字典

元组

  • 定义：元组与列表类似，但是元组的元素不能编辑;元组中只包括一个元素时，需要在元素后加逗号,否则括号会被当作运算符使用

    t1=(2018,'New',True,[1,2])
print(len(t1))        #result:4

  • 访问，使用切片

    t3=('a','b','c','d','e')
print(t3[:])             #result:('a', 'b', 'c', 'd', 'e')
print(t3[0:3])           #result:('a', 'b', 'c')
print(t3[::-1])          #result:('e', 'd', 'c', 'b', 'a')
print(t3[0:4:2])         #result:('a', 'c')

  • 运算符 + * in not in

    t4=(1,2,3)
t5=('a','b','c')
t6=t4+t5                 #新的元组
print(t6)                #result:(1, 2, 3, 'a', 'b', 'c')
print(t6[2:4]*2)         #result:(3, 'a', 3, 'a')
print('a' in t6)         #result:True
print('a' not in t6)     #result:False

  • 内置函数

    #len()
t1=('a','b','c','d','e')
print(len(t1))           #result:5

#max()、min()
t1=('a','b','c','d','e')
print(max(t1))           #result:e
print(min(t1))           #result:a

#tuple()将其他类型转换成元组
list=['a','b','c','d','e']
print(tuple(list))       #result:('a', 'b', 'c', 'd', 'e')

字典

  • 定义：另一种可变容器模型，且可存放任意类型对象，键值对形式，以逗号分隔，键必须是唯一的

    d={'name':'Liu','age':'26','birth':12.22}

  • 增删改查

    #增加
d={'name':'Liu','age':'26','birth':12.22}
d['job']='Engineer'
print(d)          #result:{'name': 'Liu', 'job': 'Engineer', 'age': '26', 'birth': 12.22}

#删除
del d['birth']    #del d["已存d在的键"]
print(d)          #result:{'name': 'Liu', 'job': 'Engineer', 'age': '26'}

#修改
d['age']='108'    #d["已存d在的键"]
print(d)          #result:{'name': 'Liu', 'job': 'Engineer', 'age': '108'}

#查询
d={'name':'Liu','age':'26','birth':12.22}
print(d['name'])                    #result:Liu
print(d.get('name'))                #result:Liu
print(d.get('job','Not find'))      #result:Not find

  • 函数

    #len()
d={'name':'Liu','age':'26','birth':12.22}
print(len(d))           #result:3

#keys()获取键
for k in d.keys():
    print(k,end="\t")   #result:name   age    birth

#values()获取值    
for v in d.values():
    print(v,end="\t")   #result:Liu    26    12.22

#items()获取每个键值对的元组
d={'name':'Liu','age':'26','birth':12.22}
for k,v in d.items():
    print(k,v,end="\t") #result:name Liu   age 26    birth 12.22

#str()转换成字符型
d={'name':'Liu','age':'26','birth':12.22}
s=str(d)
print(type(s))          #result:<class 'str'>

#clear()清空字典键值
d={'name':'Liu','age':'26','birth':12.22}
d.clear()
print(d)                 #result:{}

#copy()复制字典内容
d={'name':'Liu','age':'26','birth':12.22}
dd=d.copy()
print(dd)                #result:{'name': 'Liu', 'birth': 12.22, 'age': '26'}

#pop()没有key会抛出异常
d={'name': 'Liu', 'birth': 12.22, 'age': '26'}
d.pop('age')
print(d)                 #result:{'birth': 12.22, 'name': 'Liu'}

#popitem()随机返回并删除字典中的一对键和值
d={'name': 'Liu', 'birth': 12.22, 'age': '26'}
d.popitem()
print(d)                 #result:{'name': 'Liu', 'birth': 12.22}
