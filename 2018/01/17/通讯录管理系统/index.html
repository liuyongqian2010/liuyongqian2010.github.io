通讯录管理系统

通讯录管理系统

字典形式
    print("=======通讯录管理系统=======")
    print("1.增加姓名和手机")
    print("2.删除姓名")
    print("3.修改手机")
    print("4.查询所有用户")
    print("5.根据姓名查找手机号")
    print("6.退出")
    print("============================")
    add_list={}
    while True:
    input_val=int(input('请输入操作项：'))
    if input_val in range(1,7):
        if input_val==1:
            names=input('请输入姓名：')
            phones=input('请输入电话：')
            if phones.isdigit() and len(phones)==11:
                add_list[names] =int(phones)
            else:
                print('请正确输入电话！')
                continue
            print('姓名和电话：',add_list)
            sign=input('是否继续：y/n')
            if sign=='n':
                break
            else:
                continue
        elif input_val==2:
            print(add_list)
            sign=input('是否删除姓名：y/n')
            if sign=='y':
                del add_list
                print('已删除！')
                sign=input('是否退出y/n')
                if sign=='y':
                    break
                else:
                    continue
            else:
                print('未进行删除！')
        elif input_val==3:
            phones=input('请输入需要修改的手机：')
            phones_s=input('请输入修改后的手机：')
            if phones in str(add_list.values()):
                add_list[names]=phones_s
                print('已成功修改手机：',phones_s)
            else:
                print('修改失败！')
                sign = input('是否退出y/n')
                if sign == 'y':
                    break
                else:
                    continue
        elif input_val==4:
            print('所有用户：',list(add_list.keys()))
        elif input_val==5:
            names_reac=input('请输入姓名：')
            if names_reac in list(add_list.keys()):
                print(names_reac,'的手机为',add_list[names_reac])
                sign=input('是否继续y/n')
                if sign=='n':
                    break
                else:
                    continue
            else:
                print('未查询到该人！')
        else:
            break
    else:
        print("输入有误！")
print("谢谢使用！")
列表形式
print("=======通讯录管理系统=======")
print("1.增加姓名和手机")
print("2.删除姓名")
print("3.修改手机")
print("4.查询所有用户")
print("5.根据姓名查找手机号")
print("6.退出")
print("============================")
name_list=[]
phone_list=[]
while True:
    input_val=int(input('请输入操作项：'))
    if input_val in range(1,7):
        if input_val==1:
            name_list.append(input('请输入姓名：'))
            phone_list.append(input('请输入电话：'))
            print('姓名和电话：',name_list,phone_list)
            sign=input('是否继续：y/n')
            if sign=='n':
                break
            else:
                continue
        elif input_val==2:
            print(name_list)
            sign=input('是否删除姓名：y/n')
            if sign=='y':
                del name_list[:]
                print('已删除！')
                sign=input("是否退出y/n")
                if sign=='y':
                    break
                else:
                    continue
            else:
                print('未进行删除！')
        elif input_val==3:
            phone=(input('请输入需要修改的手机：'))
            if phone not in phone_list:
                print('请重新输入电话！')
            else:
                phone_list[(phone_list).index(phone)]=(input('请输入修改后的手机：'))
                print('已成功修改手机!')
            sign = input("是否退出y / n")
            if sign == 'y':
                break
            else:
                continue
        elif input_val==4:
            print('所有用户：',name_list[:])
        elif input_val==5:
            names_reac=input('请输入姓名：')
            if names_reac in name_list[:]:
                print(names_reac,'的手机为',phone_list[name_list.index(names_reac)])
                sign=input('是否继续y/n')
                if sign=='n':
                    break
                else:
                    continue
            else:
                print('未查询到该人！')
        else:
            break
    else:
        print("输入有误！")
print("谢谢使用！")
