python之Excel操作

使用xlwt module将数据写入Excel表格
import xlwt
book=xlwt.Workbook(encoding='utf-8',style_compression=0)
#创建一个Workbook对象，相当于创建了一个Excel文件；有两个参数：第一个设置字符编码，第二个是否压缩
sheet1=book.add_sheet('test',cell_overwrite_ok=True)
#创建一个sheet对象，一个sheet对象对应Excel文件中的一张表格
#add_sheet的第一个参数是这张表的名字，第二个参数表示覆盖单元格
sheet1.write(0,0,'EnglishName')
#其中第一个参数表示行，第二个参数表示列，第三个参数表示插入的数据
sheet1.write(1,0,'Liuyongqian')
sheet1.write(0,1,'中文名字')
sheet1.write(1,1,'刘永前')
book.save(r'e:\test1.xls')
#将以上操作保存到指定excel文件中
